Для запуска проекта выполнить команды:
cd client
npm install
npm run dev - после ввода этой команды появится ссылка http://localhost:5173/ просто ввести ее в строку браузера и сразу перекинет на сайт.

Структура проекта

client/
├── src/
│   ├── App.tsx                  # Главный компонент приложения: роли, навигация, страницы
│   ├── main.tsx                 # Точка входа React
│   ├── style.css                # Глобальные стили всего приложения
│   ├── types.ts                 # Общие TypeScript-типы
│   │
│   ├── api/                     # API-слой, подготовлен под будущий backend
│   │   ├── auth.ts              # Авторизация студента/преподавателя
│   │   ├── journal.ts           # Работа с журналом, оценками, пропусками
│   │   ├── labs.ts              # Заготовка для загрузки лабораторных
│   │   └── schedule.ts          # Получение расписания/данных главной страницы
│   │
│   ├── data/                    # Временные данные для проверки без backend
│   │   ├── mockData.ts          # Расписание, предметы, оценки, группы, ученики
│   │   ├── studentAuth.json     # Временные данные входа ученика
│   │   └── teacherAuth.json     # Временные данные входа преподавателя
│   │
│   └── pages/                   # Страницы приложения
│       ├── student/             # Раздел ученика
│       │   ├── dashboard.tsx    # Главная ученика: дедлайны, расписание, экзамены
│       │   ├── journal.tsx      # Журнал ученика: оценки, пропуски, средний балл
│       │   └── subject.tsx      # Заготовка страницы предмета
│       │
│       └── teacher/             # Раздел преподавателя
│           ├── dashboard.tsx    # Главная преподавателя: выбор группы и предмета
│           ├── journal.tsx      # Журнал преподавателя: ученики, оценки, пропуски
│           ├── schedule.tsx     # Создание ЛР/КР/ОКР и дедлайнов
│           ├── data.tsx         # Заготовка программы предмета
│           └── proverka.tsx     # Заготовка проверки лабораторных
│
├── public/                      # Статические файлы, если понадобятся
├── dist/                        # Собранный проект, не загружать на GitHub
├── node_modules/                # Зависимости, не загружать на GitHub
├── index.html                   # HTML-точка входа
├── package.json                 # Зависимости и команды запуска
├── package-lock.json            # Зафиксированные версии зависимостей
├── tsconfig.json                # Настройки TypeScript
└── vite.config.ts               # Настройки Vite
